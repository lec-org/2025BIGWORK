#include "Block.h"
#include <stdlib.h>
#include <easyx.h>

IMAGE* Block::imgs[3] = { NULL, };

int Block::blockNums[7] = { 0, 1, 2, 3, 4, 5, 6 };
int Block::currentIndex = -1;

Block::Block(int blockSize)
{

	this->size = blockSize;

	if (imgs[0] == NULL) {

		imgs[0] = new IMAGE;
		loadimage(imgs[0], _T("res/block.png"), blockSize, blockSize);
		imgs[1] = new IMAGE;
		loadimage(imgs[1], _T("res/reblock.png"), blockSize, blockSize);
		imgs[2] = new IMAGE;
		loadimage(imgs[2], _T("res/garbageblock.png"), blockSize, blockSize);

	}

	int blocks[7][4] = {
		1,3,5,7,
		2,4,5,7,
		3,4,5,6,
		3,4,5,7,
		2,3,5,7,
		3,5,6,7,
		2,3,4,5,
	};

	//随机
	blockType = getRandomBlockType();

	for (int i = 0; i < 4; i++) {
		int value = blocks[blockType][i];
		smallBlocks[i].row = value / 2;
		smallBlocks[i].col = value % 2+4;

	}

	img = imgs[0];

}

Block::Block():Block(29)
{
}

void Block::drop()
{
	for (int i = 0; i < 4; i++) {
		smallBlocks[i].row++;
	}
}

void Block::moveLeftRight(int offset)
{
	for (int i = 0; i < 4; i++) {
		smallBlocks[i].col += offset;
	}
}

void Block::rotate()
{
	Point p = smallBlocks[1];
	for (int i = 0; i < 4; i++) {
		Point tmp = smallBlocks[i];
		smallBlocks[i].col = p.col - tmp.row + p.row;
		smallBlocks[i].row = p.row + tmp.col - p.col;
	}
}

void Block::draw(int leftMargin, int topMargin)
{
	for (int i = 0; i < 4; i++) {
		int x = leftMargin + smallBlocks[i].col * size;
		int y = topMargin + smallBlocks[i].row * size;
		putimage(x, y, img);
	}
}

IMAGE* Block::getImages()
{
	return imgs[0];
}

IMAGE* Block::getPreviewImages()
{
	return imgs[1];
}

IMAGE* Block::getGarbageImage()
{
	return imgs[2];
}

Block& Block::operator=(const Block& other)
{
	if (this == &other) return *this;

	this->blockType = other.blockType;
	this->size = other.size;
	for (int i = 0; i < 4; i++) {
		this->smallBlocks[i] = other.smallBlocks[i];
	}
	return *this;
}

bool Block::blockInMap(const vector<vector<int>>& map)
{
	int rows = map.size();
	int cols = map[0].size();
	for (int i = 0; i < 4; i++) {
		if (smallBlocks[i].col < 0 || smallBlocks[i].col >= cols ||
			smallBlocks[i].row < 0 || smallBlocks[i].row >= rows ||
			map[smallBlocks[i].row][smallBlocks[i].col] != 0) return false;
	}
	return true;
}

void Block::solidify(vector<vector<int>>& map)
{
	for (int i = 0; i < 4; i++) {
		map[smallBlocks[i].row][smallBlocks[i].col] = blockType + 1;
	}
}

int Block::getBlockType()
{
	return blockType;
}

void Block::shuffleBlock()
{
	for (int i = 6; i > 0; i--)
	{
		int randPos = rand() % (i + 1);
		int temp = blockNums[i];
		blockNums[i] = blockNums[randPos];
		blockNums[randPos] = temp;
	}
}

int Block::getRandomBlockType()
{

	if (currentIndex == -1 || currentIndex >= 7)
	{
		shuffleBlock();
		currentIndex = 0;
	}
	return blockNums[currentIndex++];
}
