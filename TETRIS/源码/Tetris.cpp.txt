#include "Tetris.h"
#include <stdlib.h>
#include <time.h>
#include "Block.h"
#include "easyx.h"
#include"choose.h"

const int SPEED_NORMAL = 500;
const int SPEED_QUICK = 50;

Tetris::Tetris(int rows, int cols, int left, int top, int blockSize)
{
	this->rows = rows;
	this->cols = cols;
	this->leftMargin = left;
	this->topMargin = top;
	this->blockSize = blockSize;

	for (int i = 0; i < rows; i++) {
		vector<int>mapRow;
		for (int j = 0; j < cols; j++) {
			mapRow.push_back(0);
		}
		map.push_back(mapRow);
	}

}


void Tetris::init()
{
	delay = SPEED_NORMAL;
	loadimage(&imgBg1, _T("res/bg1.png"));  //背景1
	loadimage(&imgBeginBg, _T("res/单人教程.png"));  //教程
	loadimage(&imgReady, _T("res/单人准备.png"));  //
	for (int i = 0; i < rows; i++) {  //初始化地图
		for (int j = 0; j < cols; j++) {
			map[i][j] = 0;
		}
	}

	score = 0;
	line = 0;
	gameOver = false;
}

void Tetris::play()
{
	init();
	putimage(0, 0, &imgBeginBg);

	ExMessage msg = { 0 };
	bool p = false;

	while (1) {
		peekmessage(&msg, EX_KEY);
		if (msg.message == WM_KEYDOWN) {
			putimage(0, 0, &imgReady);
			Sleep(100);
			msg.message = 0;
			break;
		}
		msg.message = 0;
		Sleep(10);
	}

	nextBlock = new Block(blockSize);
	curBlock = nextBlock;
	nextBlock = new Block(blockSize);


	int timer = 0;
	while (1) {
		peekmessage(&msg, EX_KEY);
		if(msg.message == WM_KEYDOWN) keyEvent(msg);
		msg.message = 0;
		timer += getDelay();
		if (timer > delay) {
			timer = 0;
			update = true;
		}
		if (update) {
			update = false;
			drop();
			clearLine();
			updateWindow();
		}
		if (gameOver) {
			displayOver();
			
		}
	}
}

void Tetris::keyEvent(ExMessage msg)
{
	 bool rotateFlag = false;
	 int dx = 0;

	switch (msg.vkcode) {
	case 'W':
		rotateFlag = true;
		break;
	case 'S':
		delay = SPEED_QUICK;
		update = true;
		break;
	case 'A':
		dx = -1;
		break;
	case 'D':
		dx = 1;
		break;
	case VK_SPACE:
		dropToBottom();
		update = true;
		break;
	}

	if (rotateFlag) {
		rotate();
		updateWindow();
	}

	if (dx != 0) {
		moveLeftRight(dx);
		updateWindow();
	}
}

void Tetris::updateWindow()
{

	BeginBatchDraw();
	putimage(0, 0, &imgBg1);
	IMAGE* imgs = Block::getImages();
	drawDropPreview();
	for (int i = 0; i < rows; i++) {
		for (int j = 0; j < cols; j++) {
			if (map[i][j] == 0) continue;

			int x = j * blockSize + leftMargin;
			int y = i * blockSize + topMargin;
			putimage(x, y, imgs);
		}
	}

	curBlock->draw(leftMargin, topMargin);
	
	nextBlock->draw(823, 168);

	drawScore();

	EndBatchDraw();
}

int Tetris::getDelay()
{
	static unsigned long long lastTime = 0;
	unsigned long long currentTime = GetTickCount();

	if (lastTime == 0) {
		lastTime = currentTime;
		return 0;
	}
	else {
		int ret = currentTime - lastTime;
		lastTime = currentTime;
		return ret;
	}
	
}

void Tetris::drop()
{
	bakBlock = *curBlock;
	curBlock->drop();

	if (curBlock->blockInMap(map) == false) {
		bakBlock.solidify(map);
		delete curBlock;
		curBlock = nextBlock;
		nextBlock = new Block(blockSize);

		checkOver();
	}
	delay = SPEED_NORMAL;
}

void Tetris::clearLine()
{
	int lines = 0;
	int k = rows - 1;
	for (int i = rows - 1; i >= 0; i--) {
		int count = 0;
		for (int j = 0; j < cols; j++) {
			if (map[i][j]) {
				count++;
			}
			map[k][j] = map[i][j];
		}

		if (count < cols) {
			k--;
		}
		else {
			lines++;
		}
	}

	if (lines > 0) {
		//score
		int addScore[4] = { 10,30,60,100 };
		score += addScore[lines - 1];
		line += lines;
		update = true;
	}
}

void Tetris::moveLeftRight(int offset)
{
	bakBlock = *curBlock;
	curBlock->moveLeftRight(offset);

	if (!curBlock->blockInMap(map)) {
		*curBlock = bakBlock;
	}
}

static const Point wallKickOffsets[] = {
	{0, 0},   // 无偏移（原旋转位置）
	{1, 0},   // 右移1格
	{-1, 0},  // 左移1格
	{0, 1},   // 下移1格
	{0, -1},  // 上移1格
	{1, 1},   // 右下1格
	{-1, 1},  // 左下1格
};
// 偏移表长度
static const int kickOffsetCount = sizeof(wallKickOffsets) / sizeof(Point);

void Tetris::rotate()
{
	if (curBlock->getBlockType() == 6) return;

	bakBlock = *curBlock;
	curBlock->rotate();

	bool su = false;
	for (int i = 0; i < kickOffsetCount; i++) {
		int dx = wallKickOffsets[i].col;
		int dy = wallKickOffsets[i].row;

		curBlock->moveLeftRight(dx);
		for (int j = 0; j < 4; j++) {
			curBlock->smallBlocks[j].row += dy;
		}
		if (curBlock->blockInMap(map)) {
			su = true;
			break;
		}
		curBlock->moveLeftRight(-dx);
		for (int j = 0; j < 4; j++) {
			curBlock->smallBlocks[j].row -= dy;
		}
	}

	if (!su) {
		*curBlock = bakBlock;
	}
}

void Tetris::drawScore()
{
	char scoreText[32];
	sprintf_s(scoreText, sizeof(scoreText), "%d", score);

	char lineText[32];
	sprintf_s(lineText, sizeof(lineText), "%d", line);

	settextcolor(RGB(201, 201, 201));

	LOGFONT f;
	gettextstyle(&f);
	f.lfHeight = 50;

	f.lfQuality = ANTIALIASED_QUALITY;
	strcpy_s(f.lfFaceName, sizeof(f.lfFaceName), "Geometos");
	settextstyle(&f);
	setbkmode(TRANSPARENT);

	outtextxy(896, 704, scoreText);
	outtextxy(896, 598, lineText);
}

void Tetris::checkOver()
{
	gameOver = (curBlock->blockInMap(map) == false);
}

void Tetris::displayOver()
{
	setfillcolor(WHITE), setlinecolor(RGB(201,201,201));
	setlinestyle(PS_SOLID | PS_ENDCAP_SQUARE,3);
	fillrectangle(550, 220, 930, 620);
	outtextxy(600, 245, _T("Game Over"));

	char scoreText[32];
	sprintf_s(scoreText, sizeof(scoreText), "%d", score);
	char lineText[32];
	sprintf_s(lineText, sizeof(lineText), "%d", line);

	settextstyle(35, 0, "Geometos");
	outtextxy(576, 325, _T("Lines Cleared"));
	outtextxy(575, 365, lineText);
	outtextxy(575, 425, _T("Total Score"));
	outtextxy(575, 465, scoreText);

	setlinestyle(PS_NULL);
	setfillcolor(RGB(226, 226, 226));
	fillrectangle(564, 568, 916, 608);
	settextstyle(32, 0, "Geometos");
	settextcolor(WHITE);
	outtextxy(577, 572, _T("press r to restart"));

	ExMessage msg = { 0 };
	while (1) {
		peekmessage(&msg, EX_KEY);
		if (msg.message == WM_KEYDOWN) {
			if (msg.vkcode == 'R') {
				msg.message = 0;
				choose::modechoose();
			}
		}
		msg.message = 0;
		Sleep(10);
	}
}

void Tetris::dropToBottom()
{
	bakBlock = *curBlock;

	while (true){
		curBlock->drop();

		if (!curBlock->blockInMap(map))
		{
			*curBlock = bakBlock;
			curBlock->solidify(map);
			break;
		}

		bakBlock = *curBlock;
	}

	delete curBlock;
	curBlock = nextBlock;
	nextBlock = new Block(blockSize);

	checkOver();
	clearLine();
	delay = SPEED_NORMAL;
}

void Tetris::drawDropPreview()
{
	if (gameOver || curBlock == nullptr) return;

	Block previewBlock = *curBlock;
	Block bakPreview = previewBlock;

	while (true)
	{
		previewBlock.drop();
		if (!previewBlock.blockInMap(map))
		{
			previewBlock = bakPreview;
			break;
		}
		bakPreview = previewBlock;
	}

	IMAGE* previewImgs = Block::getPreviewImages();
	for (int i = 0; i < 4; i++)
	{
		int x = leftMargin + previewBlock.smallBlocks[i].col * blockSize;
		int y = topMargin + previewBlock.smallBlocks[i].row * blockSize;
		putimage(x, y, previewImgs);
	}
}
