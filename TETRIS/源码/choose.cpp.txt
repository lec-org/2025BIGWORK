#include <easyx.h>
#include"Tetris.h"
#include"Tetris2.h"
#include<windows.h>
#include "choose.h"


void choose::modechoose()
{
	setbkcolor(RGB(30, 30, 30));
	cleardevice();

	LOGFONT font;
	gettextstyle(&font);
	font.lfHeight = 30;
	font.lfQuality = ANTIALIASED_QUALITY;
	strcpy_s(font.lfFaceName, sizeof(font.lfFaceName), "Geometos");
	settextstyle(&font);
	setbkmode(TRANSPARENT);
	settextcolor(RGB(200, 200, 200));
	outtextxy(120, 50, "one player (press 1)");
	outtextxy(120, 100, "two players (press 2)");

	int mode = 1;
	ExMessage msg = { 0 };
	while (1) {
		peekmessage(&msg, EX_KEY);
		if (msg.vkcode == '1') {
			mode = 1;
			break;
		}
		else if (msg.vkcode == '2') {
			mode = 2;
			break;
		}
	}

	//选择模式
	if (mode == 2) {
		Tetris2 game(22, 10, 250, 856, 100, 29);  //rows, cols, left, left2, top, blockSize
		game.play();
	}
	else {
		Tetris game(22, 10, 590, 100, 29);  //rows, cols, left, top, blockSize
		game.play();
	}
}
